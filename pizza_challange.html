def sem_2(x):
    """ 2 S.E.M for binary data (sum of data x has 1 S.D. of its sqrt)"""
    if type(x) == pd.core.groupby.groupby.SeriesGroupBy:
        return 2*np.sqrt(x.mean()/x.size())
    return 2*np.sqrt(x.mean()/len(x))
